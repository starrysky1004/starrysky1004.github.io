void __fastcall censor_string(char *a1, int a2)
{
  if ( a2 > 5 && *a1 == 'C' && a1[1] == 'T' && a1[2] == 'F' && a1[3] == '{' )
    memset(a1 + 4, 'X', a2 - 5);
}
