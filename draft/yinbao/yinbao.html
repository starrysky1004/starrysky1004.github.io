yinbao部署

sudo su

更新:sudo apt update

安装vim:sudo apt install vim

安装curl:sudo apt install curl

安装git:sudo apt install git

安装docker:sudo curl -fsSL get.docker.com -o get-docker.sh

​ sudo sh get-docker.sh

​ rm get-docker.sh

安装pip:sudo apt install pip

升级requests:sudo pip install --upgrade requests

安装docker-compose:sudo pip install docker-compose

安装nginx:sudo apt install nginx

安装net-tools:sudo apt install net-tools

下载项目:git clone http://43.138.33.82:9980/yinbao/yinbao.git Yinbao

​ cp -r ./Yinbao/dist/ ./

​ cp -r ./Yinbao/yinbao/ ./

​ sudo rm -r Yinbao

创建文件夹:sudo mkdir -p /yinbao/mysql/conf.d

​ sudo mkdir -p /yinbao/mysql/log

​ sudo mkdir -p /yinbao/mysql/data

改配置:ifconfig

​ vim ./yinbao/settings.py line:28 line:93

​ sudo vim ./nginx.conf line:12 line:19

运行容器:cd yinbao

​ docker compose up -d

​ docker exec ybbackend python manage.py makemigrations

​ docker exec ybbackend python manage.py migrate

测试:curl --location 'http://43.138.33.82:5500/auth/login/' --header 'Content-Type: application/json'

cd /etc/nginx/sites-available

rm default

mv nginx.conf default

nginx -s reload

nginx -t

sudo cp dist /yinbao
