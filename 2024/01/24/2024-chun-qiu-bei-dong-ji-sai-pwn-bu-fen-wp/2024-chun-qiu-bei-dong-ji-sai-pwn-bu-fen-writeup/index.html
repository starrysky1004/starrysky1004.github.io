menu = '> '

def add(index, size):
    r.sendlineafter(menu, '1')
    r.sendlineafter(b'Index', str(index))
    r.sendlineafter(b'what size :', str(size))

def delete(index):
    r.sendlineafter(menu, '2')
    r.sendlineafter(b'Index', str(index))

def show(index):
    r.sendlineafter(menu, '3')
    r.sendlineafter(b'Index', str(index))

    r.sendlineafter(menu, '4')
    r.sendlineafter(b'Index', str(index))
